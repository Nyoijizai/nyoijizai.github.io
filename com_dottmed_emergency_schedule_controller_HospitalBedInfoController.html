作者: zhangWei

请求URL

/hospitalBed/info/add POST

请求体 application/json

{
	"currentPage":"int //当前页(分页工具类Pageable第一页下标为0)",
	"pageSize":"int //页大小",
	"pageOrders":[{
		"direction":"string //排序方式：asc | desc",
		"property":"string //排序字段"
	}],
	"pageOrdersJSON":"string //字段排序规则列表，要求JSON字符串格式，适用get请求时的字段排序规则",
	"id":"int //主键",
	"bedAreaId":"int //病床归属区域ID",
	"bedType":"int //病床类型：1 儿童， 2 男病床， 3 女病床， 4 康复， 5 隔离， 6 ICU",
	"caseHistoryNo":"string //病历编号",
	"createdCardTime":"string //建卡时间",
	"currentCourse":"string //当前病程",
	"priority":"int //优先级：0 死亡， 1 第一优先， 2 第二优先， 3 第三优先",
	"injuryCategory":"string //伤情类别",
	"diseaseReason":"string //疾病原由",
	"isInsulate":"int //是否隔离：0 否， 1 是",
	"patientName":"string //病人姓名",
	"patientAge":"string //病人年龄",
	"patientDiet":"string //饮食",
	"patientChiefComplaint":"string //主诉",
	"preDisasterSalvation":"string //灾前救助",
	"specialProtection":"string //特殊保护",
	"receptionMeasure":"string //接诊防护措施",
	"vitalSignTemperature":"string //体温",
	"vitalSignBloodOxygen":"string //血氧",
	"vitalSignPulseRate":"string //脉率",
	"vitalSignRespiratoryRate":"string //呼吸率",
	"vitalSignBloodPressure":"string //血压",
	"vitalSignStature":"string //身高",
	"vitalSignWeight":"string //体重",
	"isEventType":"int //是否事件类型：1 普通， 2 事件",
	"createMan":"int //创建人",
	"createTime":"date //创建时间"
}

返回结果

{
	"code":"int //返回码",
	"message":"string //返回消息",
	"data":{}
}
