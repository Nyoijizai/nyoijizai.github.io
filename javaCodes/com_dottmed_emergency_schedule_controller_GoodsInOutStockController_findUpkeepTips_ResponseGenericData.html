public class ResponseGenericData implements Serializable{

    private int code; //返回码
    private String message; //返回消息
    private UpkeepGoods[] data; //返回数据

    public int getCode(){
        return code;
    }

    public void setCode(int code){
        this.code = code;
    }

    public String getMessage(){
        return message;
    }

    public void setMessage(String message){
        this.message = message;
    }

    public UpkeepGoods[] getData(){
        return data;
    }

    public void setData(UpkeepGoods[] data){
        this.data = data;
    }

}

public class UpkeepGoods implements Serializable{

    private String goodsName; //物资名称
    private String goodsCode; //物资编码
    private String department; //科室
    private String boxName; //箱子名称
    private String position; //货架名称
    private String state; //状态
    private String expirationDate; //保质日期

    public String getGoodsName(){
        return goodsName;
    }

    public void setGoodsName(String goodsName){
        this.goodsName = goodsName;
    }

    public String getGoodsCode(){
        return goodsCode;
    }

    public void setGoodsCode(String goodsCode){
        this.goodsCode = goodsCode;
    }

    public String getDepartment(){
        return department;
    }

    public void setDepartment(String department){
        this.department = department;
    }

    public String getBoxName(){
        return boxName;
    }

    public void setBoxName(String boxName){
        this.boxName = boxName;
    }

    public String getPosition(){
        return position;
    }

    public void setPosition(String position){
        this.position = position;
    }

    public String getState(){
        return state;
    }

    public void setState(String state){
        this.state = state;
    }

    public String getExpirationDate(){
        return expirationDate;
    }

    public void setExpirationDate(String expirationDate){
        this.expirationDate = expirationDate;
    }

}
