public class ResponseGenericData implements Serializable{

    private int code; //返回码
    private String message; //返回消息
    private DepartmentOutGoods[] data; //返回数据

    public int getCode(){
        return code;
    }

    public void setCode(int code){
        this.code = code;
    }

    public String getMessage(){
        return message;
    }

    public void setMessage(String message){
        this.message = message;
    }

    public DepartmentOutGoods[] getData(){
        return data;
    }

    public void setData(DepartmentOutGoods[] data){
        this.data = data;
    }

}

public class DepartmentOutGoods implements Serializable{

    private String department; //科室名称
    private String percent; //百分比

    public String getDepartment(){
        return department;
    }

    public void setDepartment(String department){
        this.department = department;
    }

    public String getPercent(){
        return percent;
    }

    public void setPercent(String percent){
        this.percent = percent;
    }

}
