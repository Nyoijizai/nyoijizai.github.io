public class ResponseGenericData implements Serializable{

    private int code; //返回码
    private String message; //返回消息
    private ElectronicFenceInfo data; //返回数据

    public int getCode(){
        return code;
    }

    public void setCode(int code){
        this.code = code;
    }

    public String getMessage(){
        return message;
    }

    public void setMessage(String message){
        this.message = message;
    }

    public ElectronicFenceInfo getData(){
        return data;
    }

    public void setData(ElectronicFenceInfo data){
        this.data = data;
    }

}

public class ElectronicFenceInfo implements Serializable{

    private String createTime; //创建时间
    private int id; //主键
    private String fenceName; //围栏名称
    private int fenceFeature; //围栏功能：0 禁出， 1 禁入
    private int fenceStatus; //围栏状态：0 启动， 1 停止
    private String fenceScope; //围栏坐标范围
    private int isEventType; //是否事件类型：1 普通， 2 事件
    private int createMan; //创建人
    private String createTime; //创建时间

    public String getCreateTime(){
        return createTime;
    }

    public void setCreateTime(String createTime){
        this.createTime = createTime;
    }

    public int getId(){
        return id;
    }

    public void setId(int id){
        this.id = id;
    }

    public String getFenceName(){
        return fenceName;
    }

    public void setFenceName(String fenceName){
        this.fenceName = fenceName;
    }

    public int getFenceFeature(){
        return fenceFeature;
    }

    public void setFenceFeature(int fenceFeature){
        this.fenceFeature = fenceFeature;
    }

    public int getFenceStatus(){
        return fenceStatus;
    }

    public void setFenceStatus(int fenceStatus){
        this.fenceStatus = fenceStatus;
    }

    public String getFenceScope(){
        return fenceScope;
    }

    public void setFenceScope(String fenceScope){
        this.fenceScope = fenceScope;
    }

    public int getIsEventType(){
        return isEventType;
    }

    public void setIsEventType(int isEventType){
        this.isEventType = isEventType;
    }

    public int getCreateMan(){
        return createMan;
    }

    public void setCreateMan(int createMan){
        this.createMan = createMan;
    }

    public String getCreateTime(){
        return createTime;
    }

    public void setCreateTime(String createTime){
        this.createTime = createTime;
    }

}
