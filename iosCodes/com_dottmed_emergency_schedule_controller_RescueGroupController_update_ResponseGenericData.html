#import <Foundation/Foundation>;

@interface ResponseGenericData : NSObject

//返回码
@property (nonatomic, strong) NSInteger code;
//返回消息
@property (nonatomic, assign) NSString message;
//返回数据
@property (nonatomic, strong) RescueGroup data;


@end



#import <Foundation/Foundation>;

@interface RescueGroup : NSObject

//创建时间
@property (nonatomic, assign) NSString createTime;
//主键
@property (nonatomic, strong) NSInteger id;
//小组名称
@property (nonatomic, assign) NSString groupName;
//小组类型：1 永久， 2 临时
@property (nonatomic, strong) NSInteger groupType;
//小组状态：0 禁用，1 启用
@property (nonatomic, strong) NSInteger groupStatus;
//事件编码：事件管理表主键ID
@property (nonatomic, strong) NSInteger eventId;
//事件名称
@property (nonatomic, assign) NSString eventName;
//是否事件类型：1 普通， 2 事件
@property (nonatomic, strong) NSInteger isEventType;
//是否删除：0 未删， 1 已删
@property (nonatomic, strong) NSInteger isDelete;
//创建人(登录用户表ID)
@property (nonatomic, strong) NSInteger createMan;


@end
