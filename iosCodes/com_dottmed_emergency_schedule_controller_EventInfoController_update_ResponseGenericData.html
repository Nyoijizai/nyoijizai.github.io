#import <Foundation/Foundation>;

@interface ResponseGenericData : NSObject

//返回码
@property (nonatomic, strong) NSInteger code;
//返回消息
@property (nonatomic, assign) NSString message;
//返回数据
@property (nonatomic, strong) EventInfo data;


@end



#import <Foundation/Foundation>;

@interface EventInfo : NSObject

//创建时间
@property (nonatomic, assign) NSString createTime;
//主键
@property (nonatomic, strong) NSInteger id;
//事件名称
@property (nonatomic, assign) NSString eventName;
//事件类型：1 应急救援， 2 演练， 3 培训， 4 其他
@property (nonatomic, strong) NSInteger eventType;
//事件来源：1 WHO， 2 国家卫健委， 3 省卫健委， 4 其他有关部门
@property (nonatomic, strong) NSInteger eventSource;
//事件状态：1 进行中， 2 已结束
@property (nonatomic, strong) NSInteger eventStatus;
//维修/保养：1 维修， 2 保养
@property (nonatomic, strong) NSInteger maintenance;
//描述
@property (nonatomic, assign) NSString remark;
//创建人(登录用户表ID)
@property (nonatomic, strong) NSInteger createMan;


@end
