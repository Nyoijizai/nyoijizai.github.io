#import <Foundation/Foundation>;

@interface ResponseGenericData : NSObject

//返回码
@property (nonatomic, strong) NSInteger code;
//返回消息
@property (nonatomic, assign) NSString message;
//返回数据
@property (nonatomic, strong) SurveyInfo[] data;


@end



#import <Foundation/Foundation>;

@interface SurveyInfo : NSObject

//创建时间
@property (nonatomic, assign) NSString createTime;
//主键
@property (nonatomic, strong) NSInteger id;
//人员信息表ID
@property (nonatomic, strong) NSInteger personId;
//人员名称
@property (nonatomic, assign) NSString personName;
//联系方式
@property (nonatomic, assign) NSString personPhone;
//应急小组表ID
@property (nonatomic, strong) NSInteger groupId;
//应急小组名称
@property (nonatomic, assign) NSString groupName;
//通话开始时间
@property (nonatomic, assign) NSString talkStartTime;
//通话时长
@property (nonatomic, assign) NSString talkTime;
//勘察类型：1 拍照， 2 录像， 3 通话
@property (nonatomic, strong) NSInteger surveyType;
//事件编码：事件管理表ID
@property (nonatomic, strong) NSInteger eventId;
//事件名称
@property (nonatomic, assign) NSString eventName;
//是否删除：0 未删， 1 已删
@property (nonatomic, strong) NSInteger isDelete;
//是否事件类型：1 普通， 2 事件
@property (nonatomic, strong) NSInteger isEventType;
//文件公网访问URL地址
@property (nonatomic, assign) NSString fileUrl;


@end
