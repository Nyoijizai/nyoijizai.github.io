#import <Foundation/Foundation>;

@interface WarehouseMaterialInfo : NSObject

//主键
@property (nonatomic, strong) NSInteger id;
//物资名称
@property (nonatomic, assign) NSString materialName;
//物资编码/RFID
@property (nonatomic, assign) NSString materialCode;
//企业资产编码，如：华西资产编码
@property (nonatomic, assign) NSString assetCode;
//科室类型：1 急救/重症， 2 门诊， 3 后勤， 4 指挥， 5 重症， 6 超声， 7 清创， 8 留观， 9 药房， 10 耗材， 11 手术， 12 防疫/隔离， 13 消毒， 14 住院， 15 检验
@property (nonatomic, strong) NSInteger departmentType;
//品牌
@property (nonatomic, assign) NSString brand;
//规格型号
@property (nonatomic, assign) NSString specification;
//单位：台、件、个、袋、包等
@property (nonatomic, assign) NSString unit;
//物资状态：1 在库， 2 已出库， 3 待出库， 4 损耗， 5 报废，6 维修，7 保养
@property (nonatomic, strong) NSInteger status;
//是否装箱：0 未装箱， 1 已装箱
@property (nonatomic, strong) NSInteger isBox;
//箱子编码/箱子RFID
@property (nonatomic, assign) NSString boxCode;
//箱子名称
@property (nonatomic, assign) NSString boxName;
//货架编号
@property (nonatomic, assign) NSString rackNumber;
//货架位置：0 未知， 1 一层（下）， 2 二层（中）， 3 三层（上）， 4 四层（顶）
@property (nonatomic, strong) NSInteger rackPosition;
//生产日期
@property (nonatomic, assign) NSString productionDate;
//生产企业
@property (nonatomic, assign) NSString productionEnterprise;
//生产批号
@property (nonatomic, assign) NSString productionBatch;
//是否有保质期：0 无保质期， 1 有保质期
@property (nonatomic, strong) NSInteger isExpiration;
//过期时间
@property (nonatomic, assign) NSString expirationDate;
//创建人
@property (nonatomic, strong) NSInteger createMan;
//创建时间
@property (nonatomic, assign) NSString createTime;
//备注
@property (nonatomic, assign) NSString remark;
//物资图片列表
@property (nonatomic, strong) WarehouseMaterialImage[] materialImages;


@end



#import <Foundation/Foundation>;

@interface WarehouseMaterialImage : NSObject

//主键
@property (nonatomic, strong) NSInteger id;
//物资ID
@property (nonatomic, strong) NSInteger materialId;
//文件原名称
@property (nonatomic, assign) NSString oldFileName;
//文件新名称
@property (nonatomic, assign) NSString newFileName;
//文件后缀名
@property (nonatomic, assign) NSString fileType;
//文件访问地址
@property (nonatomic, assign) NSString fileUrl;


@end
