用户登录

作者: zhangWei

请求URL

/user/login POST

请求体 application/json

{
	"currentPage":"int //当前页(分页工具类Pageable第一页下标为0)",
	"pageSize":"int //页大小",
	"pageOrders":[{
		"direction":"string //排序方式：asc | desc",
		"property":"string //排序字段"
	}],
	"pageOrdersJSON":"string //字段排序规则列表，要求JSON字符串格式，适用get请求时的字段排序规则\r\t   格式:[{'direction':'desc','property':'id'}]   进行URL编码后传到后端",
	"id":"int //主键",
	"userName":"string //用户姓名",
	"phone":"string //联系手机",
	"loginAccount":"string //登录账号【必须】",
	"loginPassword":"string //登录密码【必须】",
	"userType":"int //账号类型：1 后台管理， 2 仓库管理， 3 应急办， 4 指挥员",
	"workNumber":"string //工号",
	"createMan":"int //创建人",
	"createTime":"date //创建时间"
}

返回结果

{
	"code":"int //返回码",
	"message":"string //返回消息",
	"data":{
		"createTime":"date //创建时间",
		"id":"int //主键",
		"userName":"string //用户姓名",
		"phone":"string //联系手机",
		"loginAccount":"string //登录账号",
		"loginPassword":"string //登录密码",
		"userType":"int //账号类型：1 后台管理， 2 仓库管理， 3 应急办， 4 指挥员",

createMan int 否 创建人
