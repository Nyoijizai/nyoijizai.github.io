#import <Foundation/Foundation>;

@interface ResponseGenericData : NSObject

//返回码
@property (nonatomic, strong) NSInteger code;
//返回消息
@property (nonatomic, assign) NSString message;
//返回数据
@property (nonatomic, strong) PageResult data;


@end



#import <Foundation/Foundation>;

@interface PageResult : NSObject

//当前页(分页工具类Pageable第一页下标为0)
@property (nonatomic, strong) NSInteger currentPage;
//页大小
@property (nonatomic, strong) NSInteger pageSize;
//总页数
@property (nonatomic, strong) NSInteger pageTotal;
//总条数
@property (nonatomic, strong) long totalNum;
//返回对象List
@property (nonatomic, strong) SysUser[] content;


@end



#import <Foundation/Foundation>;

@interface SysUser : NSObject

//创建时间
@property (nonatomic, assign) NSString createTime;
//主键
@property (nonatomic, strong) NSInteger id;
//用户姓名
@property (nonatomic, assign) NSString userName;
//联系手机
@property (nonatomic, assign) NSString phone;
//登录账号
@property (nonatomic, assign) NSString loginAccount;
//登录密码
@property (nonatomic, assign) NSString loginPassword;
//职位名称
@property (nonatomic, assign) NSString position;


@end
