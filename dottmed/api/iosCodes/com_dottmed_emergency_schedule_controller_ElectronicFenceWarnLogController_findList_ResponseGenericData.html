#import <Foundation/Foundation>;

@interface ResponseGenericData : NSObject

//返回码
@property (nonatomic, strong) NSInteger code;
//返回消息
@property (nonatomic, assign) NSString message;
//返回数据
@property (nonatomic, strong) PageResult data;


@end



#import <Foundation/Foundation>;

@interface PageResult : NSObject

//当前页(分页工具类Pageable第一页下标为0)
@property (nonatomic, strong) NSInteger currentPage;
//页大小
@property (nonatomic, strong) NSInteger pageSize;
//总页数
@property (nonatomic, strong) NSInteger pageTotal;
//总条数
@property (nonatomic, strong) long totalNum;
//返回对象List
@property (nonatomic, strong) ElectronicFenceWarnLog[] content;


@end



#import <Foundation/Foundation>;

@interface ElectronicFenceWarnLog : NSObject

//创建时间
@property (nonatomic, assign) NSString createTime;
//主键
@property (nonatomic, strong) NSInteger id;
//围栏ID
@property (nonatomic, strong) NSInteger fenceId;
//围栏名称
@property (nonatomic, assign) NSString fenceName;
//人员信息表ID
@property (nonatomic, strong) NSInteger personId;
//人员名称
@property (nonatomic, assign) NSString personName;
//联系方式
@property (nonatomic, assign) NSString personPhone;
//应急小组表ID
@property (nonatomic, strong) NSInteger groupId;
//应急小组名称
@property (nonatomic, assign) NSString groupName;
//终端设备号
@property (nonatomic, assign) NSString terminalDeviceCode;
//警告类型：1 未到场， 2 擅离， 3 闯入
@property (nonatomic, strong) NSInteger warnType;
//是否已读： 0 未读， 1 已读
@property (nonatomic, strong) NSInteger isRead;
//是否事件类型：1 普通， 2 事件
@property (nonatomic, strong) NSInteger isEventType;
//是否删除：0 未删， 1 已删
@property (nonatomic, strong) NSInteger isDelete;


@end
