public class ResponseGenericData implements Serializable{

    private int code; //返回码
    private String message; //返回消息
    private PageResult data; //返回数据

    public int getCode(){
        return code;
    }

    public void setCode(int code){
        this.code = code;
    }

    public String getMessage(){
        return message;
    }

    public void setMessage(String message){
        this.message = message;
    }

    public PageResult getData(){
        return data;
    }

    public void setData(PageResult data){
        this.data = data;
    }

}

public class PageResult implements Serializable{

    private int currentPage; //当前页(分页工具类Pageable第一页下标为0)
    private int pageSize; //页大小
    private int pageTotal; //总页数
    private long totalNum; //总条数
    private PersonnelInfo[] content; //返回对象List

    public int getCurrentPage(){
        return currentPage;
    }

    public void setCurrentPage(int currentPage){
        this.currentPage = currentPage;
    }

    public int getPageSize(){
        return pageSize;
    }

    public void setPageSize(int pageSize){
        this.pageSize = pageSize;
    }

    public int getPageTotal(){
        return pageTotal;
    }

    public void setPageTotal(int pageTotal){
        this.pageTotal = pageTotal;
    }

    public long getTotalNum(){
        return totalNum;
    }

    public void setTotalNum(long totalNum){
        this.totalNum = totalNum;
    }

    public PersonnelInfo[] getContent(){
        return content;
    }

    public void setContent(PersonnelInfo[] content){
        this.content = content;
    }

}

public class PersonnelInfo implements Serializable{

    private String createTime; //创建时间
    private int id; //主键
    private String personName; //人员姓名
    private String personPhone; //联系电话
    private int groupId; //归属小组(救援小组表ID)
    private String eventCode; //事件编码：事件管理表主键
    private String eventName; //事件名称
    private int isEventType; //是否事件类型：1 普通， 2 事件
    private int createMan; //创建人(登录用户表ID)

    public String getCreateTime(){
        return createTime;
    }

    public void setCreateTime(String createTime){
        this.createTime = createTime;
    }

    public int getId(){
        return id;
    }

    public void setId(int id){
        this.id = id;
    }

    public String getPersonName(){
        return personName;
    }

    public void setPersonName(String personName){
        this.personName = personName;
    }

    public String getPersonPhone(){
        return personPhone;
    }

    public void setPersonPhone(String personPhone){
        this.personPhone = personPhone;
    }

    public int getGroupId(){
        return groupId;
    }

    public void setGroupId(int groupId){
        this.groupId = groupId;
    }

    public String getEventCode(){
        return eventCode;
    }

    public void setEventCode(String eventCode){
        this.eventCode = eventCode;
    }

    public String getEventName(){
        return eventName;
    }

    public void setEventName(String eventName){
        this.eventName = eventName;
    }

    public int getIsEventType(){
        return isEventType;
    }

    public void setIsEventType(int isEventType){
        this.isEventType = isEventType;
    }

    public int getCreateMan(){
        return createMan;
    }

    public void setCreateMan(int createMan){
        this.createMan = createMan;
    }

}
